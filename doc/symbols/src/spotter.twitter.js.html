  9 if(!spotter)
 10     throw new Error("spotter not yet loaded!");

 15 if(!spotter.modules) spotter.modules = {};
 16 else if(typeof spotter.modules != "object")
 17     throw new Error("spotter.modules is not an object!");
 18 
 19 if(!spotter.modules.twitter) spotter.modules.twitter = {};
 20 else if(typeof spotter.modules.twitter != "object")
 21     throw new Error("spotter.modules.twitter is not an object!");
 22 
 23 /**
 24  * Required options: searchString
 25  * Other available options: ?
 26  * callback return format: {update, data}
 27  * update: true/false depending on whether there are new tweets
 28  * data: the new tweet objects themselves
 29  */
 30 spotter.modules.twitter.search = function(options)  {
 31     var refreshURL = "";
 32     var searchString = options.searchString;
 33 
 34     if(searchString === undefined || searchString === "")
 35 	throw new Error("twitter search module requires searchString to be specified as an option");
 36 
 37     this.url = function()  {
 38 	var url = 'http://search.twitter.com/search.json'
 39 	url += refreshURL != ""?refreshURL:'?q='+escape(searchString);
 40 	return url;
 41     }
 42 
 43     this.process = function(rawData)  {
 44 	var processedData = {};
 45 	refreshURL = rawData.refresh_url;
 46 	processedData.update = (rawData.results.length>0)?true:false;
 47 	processedData.data = rawData.results;
 48 	return processedData;;
 49     }
 50 };
 51 
 52 
 53 /**
 54  * Required options:
 55  * Other available options: exclude:hashtags
 56  * callback return format: {added,removed,trends}
 57  * added: new trends since the last call
 58  * removed: removed trends since the last call
 59  * trends: all trends
 60  */
 61 spotter.modules.twitter.trends = function(options)  {
 62     var lastTrends;
 63 
 64     this.url = function()  {
 65 	var url = "http://search.twitter.com/trends.json?";
 66 	if(options != undefined && options.exclude != undefined) url+="exclude="+options.exclude;
 67 	return url;
 68     }
 69 
 70     this.process = function(rawData)  {
 71 	var processedData = {};
 72 	var trends = rawData.trends;
 73 	if(lastTrends === null)  {
 74 	    processedData = {data:{"added":rawData.trends, "removed":{}, "trends":rawData.trends}};
 75 	}
 76 	else  {
 77 	    var tempArray = spotter.util.complements(rawData.trends, lastTrends);
 78 	    processedData = {data:{"added":tempArray[0],"removed":tempArray[1], "trends":rawData.trends}};
 79 	}
 80 	lastTrends = rawData.trends;
 81 	processedData.update = (processedData.data.added.length>0||processedData.data.removed.length>0)?true:false;
 82 	return processedData;
 83     }
 84 }
