  8 /**
  9  * Returns an array of integers that represent
 10  * the indices of the elements of b in the elements
 11  * of a.  Currently assumes these are trend objects.
 12  * Also assumes that all elements in a and b are uniq
 13  * (i.e. they are sets)
 14  *
 15  * For example
 16  *
 17  * a:      ["a","b","c","d"]
 18  * b:      ["c","b","d","f"]
 19  * result: [-1 , 1 , 0 , 2 ]  
 20  *
 21  * @param {Array} An array of length n
 22  * @param {Array} An array of length n
 23  *
 24  * TODO: make this more general
 25  * TODO: make this private
 26  */
 27 spotter.util.changes = function(a,b)  {
 28     /*a = [{'name':'a'},{'name':'b'},{'name':'c'},{'name':'d'}];
 29       b = [{'name':'c'},{'name':'b'},{'name':'d'},{'name':'f'}];*/
 30     
 31     var result = new Array();
 32     var indices = new Object();
 33     for(var i in b)
 34 	indices[b[i]]==undefined?indices[b[i]['name']]=parseInt(i):null;
 35     for(var i in a)
 36 	result[i] = indices[a[i]['name']]==undefined?-1:indices[a[i]['name']];
 37     return result;
 38 }

 40 /**
 41  * returns an array of arrays.  the first
 42  * are the elements in a that are not in b
 43  * and the second are the elements in b that
 44  * are not in a.
 45  *
 46  * For now this assumes a trends object
 47  *
 48  * TODO: make private
 49  * TODO: make more general (for arbitrary arrays)
 50  * TODO: use the changes algorithm as a subroutine
 51  *
 52  */
 53 spotter.util.complements = function(a, b)  {
 54     var counts = new Object();
 55     var aMinusB = new Array();
 56     var bMinusA = new Array();
 57     for(var i in a)
 58 	counts[a[i]]==undefined?counts[a[i]['name']]=i:null;
 59     for(var j in b)
 60 	counts[b[j]['name']]==null?bMinusA.push(b[j]):counts[b[j]['name']]=-1;
 61     for(var k in counts)
 62 	counts[k] >= 0?aMinusB.push(a[counts[k]]):null;
 63     return [aMinusB,bMinusA];
 64 }
