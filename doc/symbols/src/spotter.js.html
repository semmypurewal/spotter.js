 12 /**
 13  * @namespace
 14  * The namespace of this library.
 15  */
 16 spotter = {};
 17 
 18 /**
 19  * @namespace
 20  * The module namespace
 21  */
 22 spotter.modules = {};
 23 
 24 /**
 25  * Construct a Spotter object of the specified type with the specified
 26  * options.  See the specific module documentation for available
 27  * options.
 28  *
 29  * @constructor
 30  * @param {String} type the type the module type associated witht this spotter, e.g. "twitter.search"
 31  * @param {Object} options a hash of options for the appropriate module, e.g. {searchString: "Justing Beiber"}
 32  * @throws {Error} An error is thrown if there is a problem loading the module
 33  */
 34 spotter.Spotter = function(type, options)  {
 35     this.instanceCount = this.instanceCount+1;
 36     var varName =  "so"+this.instanceCount+""+Math.floor(Math.random()*100);
 37     var lastCallReturned = true;
 38     var lastScriptTag = null;
 39     var observers = [];
 40     var intervalTimer = null;
 41     var module;
 42 
 43     window["spotter"][varName] = this;
 44 
 45     try  {
 46 	module = new (window["spotter"]["modules"][type.split(".")[0]][type.split(".")[1]])(options);
 47     } catch(e)  {
 48 	throw new Error("Spotter: Module " + type + " not found! (Did you remember to include it via a script tag?)");
 49     }
 50 
 51     if(!module.url || !module.process)  {
 52 	throw new Error("Spotter: spotter.modules."+type+" is invalid.  (Does it return an object with url and process methods?)");
 53     }
 54 
 55 
 56     /**
 57      * Begin spotting.
 58      *
 59      * @param {Number} seconds optional parameter represeting the number of seconds
 60      *        between each request.  If this parameter is not provided, the
 61      *        function performs a single request.
 62      *
 63      * TODO: set up a time out so that if the last request doesn't return 
 64      *       the remaining requests are not blocked
 65      *
 66      * TODO: get rid of the number of seconds between requests, let that be
 67      *       handled by the appropriate module
 68      *
 69      */
 70     this.spot = function(seconds)  {
 71 	var url;
 72 	
 73 	if((!seconds || seconds < 1) && lastCallReturned)  {
 74 	    url = module.url();
 75 	    if(url instanceof Object && url.callbackParam !== undefined)  {
 76 		url = url.url+'&'+url.callbackParam+'=spotter.'+varName+'.callback';
 77 	    }
 78 	    else  {
 79 		url += '&callback=spotter.'+varName+'.callback';
 80 	    }
 81 	    url += '&random='+Math.floor(Math.random()*10000);  //add random number to help avoid caching in safari and chrome
 82 	    request(url);
 83 	}
 84 	else  {
 85 	    this.spot();
 86 	    var obj = this;
 87 	    intervalTimer = setInterval(function() { obj.spot(); }, seconds*1000);
 88 	}
 89     }
 90     
 91     /**
 92      * Receives the response from the ajax request and send it
 93      * to the appropriate module for processing.  Removes the
 94      * defunct script tag from the DOM.  Notifies observers if
 95      * the module determines there is new data.
 96      *
 97      * @private 
 98      * @param {Object} rawData Unprocessed data direct from the API
 99      */
100     this.callback  = function(rawData)  {
101 	var processedData = module.process(rawData); //send the raw data to the module for processing
102 	//now the processedData has an 'update' attribute and a 'data' attribute
103 	if(processedData.update) notifyObservers(processedData.data);
104 
105 	//here is where we need to set up the next call by getting the delay from the module
106 
107 	lastCallReturned = true;
108     }
109 
110 
111     /**
112      * Stops this spotter if it is currently spotting.
113      *
114      * @throws Error An error is thrown if you try to stop a stopped spotter
115      */
116     this.stop = function()  {
117 	if(intervalTimer === null)  {
118 	    throw new Error("Spotter: You can't stop a stopped spotter!");
119 	}
120 	else  {
121 	    var head = document.getElementsByTagName("head");
122 	    if(lastScriptTag != null) head[0].removeChild(lastScriptTag);
123 	    clearInterval(intervalTimer);
124 	}
125     }
126 
127     /**
128      * Function that actually makes the request.
129      *
130      * @private
131      * @param {String} url the json request URL
132      */
133     var request = function(url)  {
134 	var head = document.getElementsByTagName("head");
135 	var script = document.createElement('script');
136 	script.id = varName+'_'+'request';
137 	script.type = 'text/javascript';
138 	script.src = url;
139 	if(lastScriptTag !== null) head[0].removeChild(lastScriptTag);
140 	head[0].appendChild(script);
141 	lastScriptTag = script;
142     }
143     
144     /********** OBSERVABLE MIXIN ***************/
145     /**
146      * Register an observer with this object
147      *
148      * @param {Object} observer this object will be notified when new data is available
149      * @throws TypeError a TypeError is thrown if the parameter does not implement notify
150      */
151     spotter.Spotter.prototype.register = function(observer) {
152 	if(observer.notify !== undefined && typeof observer.notify === 'function')
153 	    observers.push(observer);
154 	else
155 	    throw new TypeError('Observer must implement a notify method.');
156     }
157     
158     /**
159      * Notify this Observable's observers
160      *
161      * @private
162      * @param {Object} data that will be sent to the observers
163      */
164     var notifyObservers = function(data)  {
165 	for(var i in observers)
166 	    observers[i].notify(data);
167     }
168     /********** END OBSERVABLE MIXIN ***************/
169 }//end spotter constructor

178 spotter.Spotter.prototype.instanceCount = 0;
