  6 if(!spotter)
  7     throw new Error("spotter not yet loaded!");
  8 
  9 if(!spotter.util)
 10     throw new Error("spotter.util not yet loaded!");
 11 
 12 if(!spotter.modules) spotter.modules = {};
 13 else if(typeof spotter.modules != "object")
 14     throw new Error("spotter.modules is not an object!");
 15 
 16 if(!spotter.modules.twitpic) spotter.modules.twitpic = {};
 17 else if(typeof spotter.modules.twitpic != "object")
 18     throw new Error("spotter.modules.twitpic is not an object!");
 19 
 20 /**
 21  * Required options: searchString
 22  * Other available options: ?
 23  * callback return format: {update, data}
 24  *
 25  * There is no twitpic API, this is a modification of the 
 26  * twitter search API
 27  *
 28  * In addition to the normal twitter API response each object
 29  * includes the following:
 30  *
 31  * twitpic_url
 32  * twitpic_thumbnail_url
 33  * twitpic_mini_url
 34  *
 35  * update: true/false depending on whether there are new tweets
 36  * data: the tweet objects themselves
 37  */
 38 spotter.modules.twitpic.search = function(options)  {
 39     var refreshURL = "";
 40     var searchString = options.searchString;
 41 
 42     if(searchString === undefined || searchString === "")
 43 	throw new Error("twitpic search module requires searchString to be specified as an option");
 44 
 45     this.url = function()  {
 46 	var url = 'http://search.twitter.com/search.json'
 47 	url += refreshURL != ""?refreshURL:'?q='+escape(searchString)+"+twitpic";
 48 	return url;
 49     }
 50 
 51     this.process = function(rawData)  {
 52 	var processedData = {};
 53 	var i;
 54 	var rematch;
 55 	var twitpic_id;
 56 	refreshURL = rawData.refresh_url;
 57 	processedData.update = (rawData.results.length>0)?true:false;
 58 
 59 	//process rawData and put it in processedData
 60 	processedData.data = [];
 61 	for(i in rawData.results)  {
 62 	    //put the processed version of the raw data in the 
 63 	    //processed data array
 64 	    rematch = /http\:\/\/twitpic.com\/(\w+)/.exec(rawData.results[i].text);
 65 	    if(rematch!== null && !rawData.results[i].text.match(new RegExp("^RT")))  {  //ignore retweets
 66 		twitpic_id = rematch[1];
 67 		rawData.results[i].twitpic_url = "http://twitpic.com/"+twitpic_id;
 68 		rawData.results[i].twitpic_full_url = "http://twitpic.com/show/full/"+twitpic_id;
 69 		rawData.results[i].twitpic_thumb_url = "http://twitpic.com/show/thumb/"+twitpic_id;
 70 		rawData.results[i].twitpic_mini_url = "http://twitpic.com/show/mini/"+twitpic_id;
 71 		processedData.data.push(rawData.results[i]);
 72 	    }
 73 	}
 74 	return processedData;
 75     }
 76 };
