  9 if(!spotter)
 10     throw new Error("spotter not yet loaded!");
 11 
 12 if(!spotter.modules) spotter.modules = {};
 13 else if(typeof spotter.modules != "object")
 14     throw new Error("spotter.modules is not an object!");
 15 
 16 if(!spotter.modules.flickr) spotter.modules.flickr = {};
 17 else if(typeof spotter.modules.flickr != "object")
 18     throw new Error("spotter.modules.flickr is not an object!");
 19 
 20 spotter.modules.flickr.search = function(options)  {
 21     if(options == undefined || options.api_key == undefined || (options.searchString == undefined && options.tags == undefined))
 22 	throw new Error("flickr module requires an api_key and a searchString or tags to be defined as an option");
 23 
 24     var api_key = options.api_key;
 25     var searchString = options.searchString;
 26     var tags = options.tags;
 27     
 28     this.url = function()  {
 29 	var url = 'http://api.flickr.com/services/rest/?method=flickr.photos.search';
 30 	url += '&api_key='+api_key+'&format=json&content_type=1';
 31 	if(tags != undefined) url+= '&tags='+escape(tags);
 32 	if(searchString != undefined) url+= '&text='+escape(searchString);
 33 	return {url:url, callbackParam:"jsoncallback"};
 34     }
 35 
 36     this.process = function(rawData)  {
 37 	var processedData = {};
 38 	var photos = rawData.photos.photo;
 39 	for(i in photos)  {
 40 	    photos[i].url = buildPhotoURL(photos[i]);
 41 	    photos[i].user_url = "http://www.flickr.com/"+photos[i].owner;
 42 	}
 43 	processedData.update = (photos.length>0)?true:false;	
 44 	processedData.data = photos;
 45 	return processedData;
 46     }
 47 
 48     /** private method that builds a photo URL from a photo object **/
 49     var buildPhotoURL = function(photo)  {
 50 	var u = "http://farm" + photo.farm + ".static.flickr.com/"+photo.server+"/"+ photo.id + "_" + photo.secret + ".jpg";
 51 	return u;
 52     }
